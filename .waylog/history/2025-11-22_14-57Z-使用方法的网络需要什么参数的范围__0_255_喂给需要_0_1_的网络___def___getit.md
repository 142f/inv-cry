        assert opt.data_label in ["train", "val"]
        self.data_label = opt.data_label
        self.real_list = utils.get_list(opt.real_list_path)
        self.fake_list = utils.get_list(opt.fake_list_path)
        self.label_dict = dict()
        for i in self.real_list:
            self.label_dict[i] = 0
        for i in self.fake_list:
            self.label_dict[i] = 1
        self.total_list = self.real_list + self.fake_list

        self.targets = [self.label_dict[path] for path in self.total_list]

    def __len__(self):
        return len(self.total_list)

    def __getitem__(self, idx):
        img_path = self.total_list[idx]
        label = self.label_dict[img_path]
        
        # 读取图像
        img_cv = cv2.imread(img_path)
        
        # 错误处理：如果图像读取失败
        if img_cv is None:
            print(f"警告: 无法读取图像 {img_path}")
            # 返回默认值，避免训练中断
            img = torch.zeros(3, 1120, 1120)
            crops = [[torch.zeros(3, 224, 224) for _ in range(5)], 
                     [torch.zeros(3, 224, 224) for _ in range(5)], 
                     [torch.zeros(3, 224, 224) for _ in range(5)]]
            return img, crops, label
        
        # 【优化：颜色转换和格式处理】
        # OpenCV 默认是 BGR，必须转 RGB
        img_cv = cv2.cvtColor(img_cv, cv2.COLOR_BGR2RGB)
        
        # 【性能优化】：使用 ToTensor() 替代手动除法和permute
        # ToTensor() 在底层 C++ 实现中处理 uint8->float 和 /255 操作，效率更高
        img = transforms.ToTensor()(img_cv)
        
        # 归一化
        img = transforms.Normalize(mean=[0.48145466, 0.4578275, 0.40821073],
                                     std=[0.26862954, 0.26130258, 0.27577711])(img)
        
        # 【修正 2：裁剪逻辑】
        # 原代码: i:i+500 (只移动1像素)
        # 新代码: i*500 : i*500 + 500 (每次移动500像素，分块: 0-500, 500-1000...)
        crops = [[transforms.Resize((224, 224))(img[:, 500:, i*500 : i*500 + 500]) for i in range(5)], [], []]
        
        crop_idx = [(28, 196), (61, 163)]
        for i in range(len(crops[0])):
            # Level 1 (Zoom)
            crops[1].append(transforms.Resize((224, 224))
                            (crops[0][i][:, crop_idx[0][0]:crop_idx[0][1], crop_idx[0][0]:crop_idx[0][1]]))
            # Level 2 (Macro)
            crops[2].append(transforms.Resize((224, 224))
                            (crops[0][i][:, crop_idx[1][0]:crop_idx[1][1], crop_idx[1][0]:crop_idx[1][1]]))
        
        # 注意：这里将整个长图 resize 成了方形，如果需要保留原始比例请留意
        img = transforms.Resize((1120, 1120))(img)

        return img, crops, label

def validate(model, loader, gpu_id):
    device = torch.device(f"cuda:{gpu_id[0]}" if torch.cuda.is_available() else "cpu")
    model.eval() # 确保模型处于验证模式

    y_true, y_pred = [], []
    
    with torch.no_grad():
        # 使用 tqdm 包装数据加载器

            # 优化：non_blocking=True 配合 pin_memory 加速数据传输
            img_tens = img.to(device, non_blocking=True)
            # 处理嵌套列表的 tensor 转移
            crops_tens = [[t.to(device, non_blocking=True) for t in sublist] for sublist in crops]

        return 0, 0, 0, 0, 0

    # 2. 计算 AUC (最高优先级)
    try:
        auc = roc_auc_score(y_true, y_pred_prob)
    except ValueError:
        auc = 0.0

    # 3. 计算 AP (次高优先级)
    ap = average_precision_score(y_true, y_pred_prob)

    # 4. 寻找最佳阈值 (Best Accuracy)
    # 既然 AUC > ACC，说明我们关心分布。默认 0.5 阈值可能不是最优的。
    # 这里利用 ROC 曲线寻找 Youden's J statistic (TPR - FPR) 最大的点作为最佳阈值

    # Youden's J index = TPR - FPR = TPR - (1 - TNR)
    j_scores = tpr_curve - fpr_curve
    best_idx = np.argmax(j_scores)
    best_threshold = thresholds[best_idx]
    
    # 使用最佳阈值计算二值化结果
    y_pred_binary = np.where(y_pred_prob >= best_threshold, 1, 0)
    
    # 5. 计算 ACC 及 混淆矩阵
    acc = accuracy_score(y_true, y_pred_binary)
    cm = confusion_matrix(y_true, y_pred_binary)
    
    # 防止除零错误
    tn, fp, fn, tp = cm.ravel()
    fnr = fn / (fn + tp) if (fn + tp) > 0 else 0
    fpr = fp / (fp + tn) if (fp + tn) > 0 else 0

    print(f"Conf Matrix: [TN:{tn} FP:{fp} | FN:{fn} TP:{tp}]")
    print(f"Best Threshold found: {best_threshold:.4f} (Default is 0.5)")
